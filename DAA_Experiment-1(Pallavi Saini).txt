/*
1. Recurrence relation: T(n)=3T(n/2)+nlogn+2n+2
2. Master Theorem: a=3, b=2, k=1, p=1
        T(n)=O(n^(log(base 2)3))
3. Operations=88
   Depth=3
*/

#include <bits/stdc++.h>
using namespace std;

int ops = 0;
int maxdph=0;
void complexRec(int n, int dph=1) {
    maxdph = max(maxdph, dph);
   if (n <= 2) {
       return;
   }


   int p = n;
   while (p > 0) {
       vector<int> temp(n);
       for (int i = 0; i < n; i++) {
           temp[i] = i ^ p;
           ops++;
       }
       p >>= 1;
   }


   vector<int> small(n);
   for (int i = 0; i < n; i++) {
       small[i] = i * i;
       ops++;
   }


   if (n % 3 == 0) {
       reverse(small.begin(), small.end());
   } else {
       reverse(small.begin(), small.end());
   }


   complexRec(n / 2, dph+1);
   complexRec(n / 2, dph+1);
   complexRec(n / 2, dph+1);
}
int main(){
    complexRec(8);
    cout<<ops<<endl;
    cout<<maxdph<<endl;

}
